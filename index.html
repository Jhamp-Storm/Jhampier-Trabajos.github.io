Setup Supabase
1. Crear proyecto en supabase.com Ve a supabase.com → New Project. Copia la Project URL y la anon key.
2. Tabla de usuarios (login) -- Tabla de usuarios con roles y contraseña hasheada CREATE TABLE users ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), username TEXT UNIQUE NOT NULL, full_name TEXT, password_hash TEXT NOT NULL, role TEXT NOT NULL CHECK (role IN ('admin','viewer')), created_at TIMESTAMPTZ DEFAULT NOW() ); -- Insertar el ÚNICO administrador (cambia la contraseña) -- Usar pgcrypto para hashear: crypt('TU_PASS', gen_salt('bf')) INSERT INTO users (username, full_name, password_hash, role) VALUES ('newkevingc@gmail.com', 'Administrador', crypt('kohana160', gen_salt('bf')), 'admin'); -- Habilitar pgcrypto si no está activo CREATE EXTENSION IF NOT EXISTS pgcrypto;
3. Tablas separadas por unidad -- Tabla Unidad 1 CREATE TABLE unit1_records ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), name TEXT NOT NULL, email TEXT NOT NULL, department TEXT, file_name TEXT, file_content TEXT, created_by TEXT, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); -- Tabla Unidad 2 CREATE TABLE unit2_records ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), name TEXT NOT NULL, email TEXT NOT NULL, department TEXT, file_name TEXT, file_content TEXT, created_by TEXT, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); -- Tabla Unidad 3 CREATE TABLE unit3_records ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), name TEXT NOT NULL, email TEXT NOT NULL, department TEXT, file_name TEXT, file_content TEXT, created_by TEXT, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); -- Tabla Unidad 4 CREATE TABLE unit4_records ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), name TEXT NOT NULL, email TEXT NOT NULL, department TEXT, file_name TEXT, file_content TEXT, created_by TEXT, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); -- Índices para búsquedas rápidas en cada tabla CREATE INDEX idx_u1_name ON unit1_records(name); CREATE INDEX idx_u2_name ON unit2_records(name); CREATE INDEX idx_u3_name ON unit3_records(name); CREATE INDEX idx_u4_name ON unit4_records(name);
4. Row Level Security (RLS) -- Habilitar RLS en todas las tablas ALTER TABLE users ENABLE ROW LEVEL SECURITY; ALTER TABLE unit1_records ENABLE ROW LEVEL SECURITY; ALTER TABLE unit2_records ENABLE ROW LEVEL SECURITY; ALTER TABLE unit3_records ENABLE ROW LEVEL SECURITY; ALTER TABLE unit4_records ENABLE ROW LEVEL SECURITY; -- Políticas: lectura pública (anon puede leer) CREATE POLICY "Public read unit1" ON unit1_records FOR SELECT USING (true); CREATE POLICY "Public read unit2" ON unit2_records FOR SELECT USING (true); CREATE POLICY "Public read unit3" ON unit3_records FOR SELECT USING (true); CREATE POLICY "Public read unit4" ON unit4_records FOR SELECT USING (true); -- Escritura: solo service_role (controlada desde JS con anon key -- más validación de rol en la app) -- Alternativa: usar una función RPC con SECURITY DEFINER
5. Función de login segura (RPC) -- Crear función para verificar contraseña CREATE OR REPLACE FUNCTION verify_login(p_username TEXT, p_password TEXT) RETURNS TABLE(id UUID, username TEXT, full_name TEXT, role TEXT) AS $$ BEGIN RETURN QUERY SELECT u.id, u.username, u.full_name, u.role FROM users u WHERE u.username = p_username AND u.password_hash = crypt(p_password, u.password_hash); END; $$ LANGUAGE plpgsql SECURITY DEFINER; -- Función para registrar visualizador CREATE OR REPLACE FUNCTION register_viewer( p_username TEXT, p_password TEXT, p_full_name TEXT ) RETURNS TEXT AS $$ BEGIN -- Verificar que no exista IF EXISTS (SELECT 1 FROM users WHERE username = p_username) THEN RETURN 'username_taken'; END IF; INSERT INTO users (username, full_name, password_hash, role) VALUES (p_username, p_full_name, crypt(p_password, gen_salt('bf')), 'viewer'); RETURN 'ok'; END; $$ LANGUAGE plpgsql SECURITY DEFINER;
