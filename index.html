CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  username TEXT UNIQUE NOT NULL,
  full_name TEXT,
  password_hash TEXT NOT NULL,
  role TEXT NOT NULL CHECK (role IN ('admin','viewer')),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Insertar el ÚNICO administrador (cambia la contraseña)
-- Usar pgcrypto para hashear: crypt('TU_PASS', gen_salt('bf'))

-- Habilitar pgcrypto si no está activo
CREATE EXTENSION IF NOT EXISTS pgcrypto;

        5. Función de login segura (RPC)
        -- Crear función para verificar contraseña
CREATE OR REPLACE FUNCTION verify_login(p_username TEXT, p_password TEXT)
RETURNS TABLE(id UUID, username TEXT, full_name TEXT, role TEXT) AS $$
BEGIN
  RETURN QUERY
  SELECT u.id, u.username, u.full_name, u.role
  FROM users u
  WHERE u.username = p_username
    AND u.password_hash = crypt(p_password, u.password_hash);
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Función para registrar visualizador
CREATE OR REPLACE FUNCTION register_viewer(
  p_username TEXT,
  p_password TEXT,
  p_full_name TEXT
) RETURNS TEXT AS $$
BEGIN
  -- Verificar que no exista
  IF EXISTS (SELECT 1 FROM users WHERE username = p_username) THEN
    RETURN 'username_taken';
  END IF;
  INSERT INTO users (username, full_name, password_hash, role)
  VALUES (p_username, p_full_name, crypt(p_password, gen_salt('bf')), 'viewer');
  RETURN 'ok';
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
