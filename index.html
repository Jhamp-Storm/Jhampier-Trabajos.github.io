Setup Supabase

      
    

    


      

        1. Crear proyecto en supabase.com
        Ve a supabase.com → New Project. Copia la URL y la anon key.
      


      

        2. Crear tablas en el SQL Editor
        -- Tabla de usuarios con roles
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  username TEXT UNIQUE NOT NULL,
  role TEXT NOT NULL CHECK (role IN ('admin','viewer')),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Tabla de registros por unidad
CREATE TABLE records (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  unit TEXT NOT NULL,
  name TEXT NOT NULL,
  email TEXT NOT NULL,
  department TEXT,
  file_name TEXT,
  file_content TEXT,
  created_by TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Índice para búsquedas rápidas
CREATE INDEX idx_records_unit ON records(unit);
CREATE INDEX idx_records_name ON records(name);
      


      

        3. Activar Row Level Security (RLS)
        -- Habilitar RLS
ALTER TABLE records ENABLE ROW LEVEL SECURITY;
ALTER TABLE users ENABLE ROW LEVEL SECURITY;

-- Política: todos leen registros (anon)
CREATE POLICY "Read records"
  ON records FOR SELECT
  USING (true);

-- Política: solo admins crean/editan/eliminan
-- (se controla desde el backend con service_role key)
      


      

        4. Datos de prueba para users
        INSERT INTO users (username, role) VALUES
  ('admin1', 'admin'),
  ('viewer1', 'viewer'),
  ('admin_demo', 'admin'),
  ('viewer_demo', 'viewer');
      


      

        5. Conectar en este archivo HTML
        // Reemplaza en el JS (línea SUPABASE CONFIG):
const SUPABASE_URL = 'https://xxxx.supabase.co';
const SUPABASE_KEY = 'tu-anon-key-aqui';
      


      

        ✓ Con esto tendrás:
        Autenticación por username → rol → acceso diferenciado. Admins CRUD completo. Visualizadores solo lectura. Datos persistentes en la nube.
