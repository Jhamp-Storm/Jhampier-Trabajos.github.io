Setup Supabase
1. Crear proyecto en supabase.com Ve a supabase.com → New Project. Copia la Project URL y la anon key.
2. Tabla de usuarios (login) CREATE TABLE users ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), username TEXT UNIQUE NOT NULL, full_name TEXT, password_hash TEXT NOT NULL, role TEXT NOT NULL CHECK (role IN ('admin','viewer')), created_at TIMESTAMPTZ DEFAULT NOW() ); INSERT INTO users (username, full_name, password_hash, role) VALUES ('newkevingc@gmail.com', 'Administrador', crypt('kohana160', gen_salt('bf')), 'admin'); CREATE EXTENSION IF NOT EXISTS pgcrypto;
3. Tablas por unidad CREATE TABLE unit1_records ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), name TEXT NOT NULL, email TEXT NOT NULL, department TEXT, file_name TEXT, file_content TEXT, created_by TEXT, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); -- Repetir para unit2_records, unit3_records, unit4_records
4. Row Level Security (RLS) ALTER TABLE unit1_records ENABLE ROW LEVEL SECURITY; CREATE POLICY "Public read unit1" ON unit1_records FOR SELECT USING (true);
5. Función de login RPC CREATE OR REPLACE FUNCTION verify_login(p_username TEXT, p_password TEXT) RETURNS TABLE(id UUID, username TEXT, full_name TEXT, role TEXT) AS $$ BEGIN RETURN QUERY SELECT u.id, u.username, u.full_name, u.role FROM users u WHERE u.username = p_username AND u.password_hash = crypt(p_password, u.password_hash); END; $$ LANGUAGE plpgsql SECURITY DEFINER;
